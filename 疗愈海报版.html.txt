        // UI 交互逻辑
        const toggleBtn = document.getElementById('ui-toggle-btn');
        const controlPanel = document.getElementById('control-panel');
        
        toggleBtn.addEventListener('click', () => { controlPanel.classList.toggle('active'); });

        document.addEventListener('click', (e) => {
            if (!controlPanel.contains(e.target) && e.target !== toggleBtn && !toggleBtn.contains(e.target)) {
                controlPanel.classList.remove('active');
            }
        });

        // P5.js 逻辑
        let springLines = [];
        let butterflies = [];
        const numLines = 45;

        let countSlider, forceSlider, sizeSlider;
        let countLabel, forceLabel, sizeLabel;

        function setup() {
            let canvas = createCanvas(windowWidth, windowHeight);
            canvas.parent('canvas-container');
            pixelDensity(displayDensity());
            initLines();

            countSlider = document.getElementById('slider-count');
            forceSlider = document.getElementById('slider-force');
            sizeSlider = document.getElementById('slider-size');
            
            countLabel = document.getElementById('val-count');
            forceLabel = document.getElementById('val-force');
            sizeLabel = document.getElementById('val-size');

            updateButterflies(parseInt(countSlider.value));
        }

        function draw() {
            background('#FAFAFA');

            let currentTargetCount = parseInt(countSlider.value);
            let currentWaveForce = parseFloat(forceSlider.value);
            let currentScale = parseFloat(sizeSlider.value);

            countLabel.innerText = currentTargetCount;
            forceLabel.innerText = currentWaveForce.toFixed(2);
            sizeLabel.innerText = currentScale.toFixed(1) + 'x';

            updateButterflies(currentTargetCount);

            for (let line of springLines) {
                line.update(currentWaveForce);
                line.display();
            }

            for (let b of butterflies) {
                b.update();
                b.display(currentScale);
            }
        }

        function windowResized() {
            resizeCanvas(windowWidth, windowHeight);
            initLines();
        }

        function initLines() {
            springLines = [];
            let spacing = height / (numLines + 1);
            for (let i = 0; i <= numLines; i++) {
                springLines.push(new SpringLine(i * spacing));
            }
        }

        function updateButterflies(target) {
            while (butterflies.length < target) { butterflies.push(new ExquisiteButterfly()); }
            while (butterflies.length > target) { butterflies.pop(); }
        }

        class SpringLine {

                this.baseY = y;
                this.points = [];

                    let x = map(i, 0, this.segments, 0, width);
                    this.points.push({ x: x, y: y, vy: 0 });
                }
